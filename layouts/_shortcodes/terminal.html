{{- /* terminal shortcode — 글 본문에 터미널 창 디자인 블록을 삽입한다(정적, 실제 기능 없음). 사용법: {{< terminal title="ken@blog: ~" >}} $ whoami HYEONJAE LEE # 아래는 명령 출력 $ kubectl get pods NAME READY STATUS web 1/1 Running {{< /terminal >}} 규칙: - "$ " 로 시작하는 줄 → 프롬프트 + 명령어(초록 강조) - "# " 로 시작하는 줄 → 주석(회색) - 그 외 줄 → 명령 출력(그대로 표시) 파라미터: - title : 창 상단 제목(기본값 "bash") */ -}} {{- $title := .Get "title" | default "bash" -}} {{- $lines := split (trim .Inner "\n\r") "\n" -}} {{- $out := slice -}} {{- range $lines -}} {{- $l := strings.TrimRight "\r" . -}} {{- if eq $l "$" -}} {{- $out = $out | append `$` -}} {{- else if hasPrefix $l "$ " -}} {{- $cmd := htmlEscape (strings.TrimPrefix "$ " $l) -}} {{- $out = $out | append (printf `$ %s` $cmd) -}} {{- else if hasPrefix $l "# " -}} {{- $out = $out | append (printf `%s` (htmlEscape $l)) -}} {{- else -}} {{- $out = $out | append (htmlEscape $l) -}} {{- end -}} {{- end -}}
{{ $title }}
{{ delimit $out "\n" | safeHTML }}
